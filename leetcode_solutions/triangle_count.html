Triangle Count

    /**
     * @param S: A list of integers.
     * @return Number of triangles these integers can form.
     * @comlexity O(nlog(n))
     */
    public int triangleCount(int S[]) {
        int rv = 0;
        Arrays.sort(S);//nlog(n)
        for(int i = 2; i < S.length; i++) {
            int low = 0;
            int high = i - 1;
            int longestEdge = S[i];
            while(low < high) {
                if(S[low] + S[high] > longestEdge) {
                    rv += high-low;
                    high--;
                }else {
                    low++;
                }
            }
        }
        return rv;
    }
