Lowest Common Ancestor of a Binary Search Tree

    /**
     * LeetCode 235
     * @param root - a [binary search tree].
     * @param p
     * @param q
     * @return Lowest Common Ancestor of p and q in BST root.
     */
    public static TreeNode lowestCommonAncestorBST(TreeNode root, TreeNode p, TreeNode q) {
        return ((root.val - p.val) * (root.val - q.val)) <= 0 ? root : 
            (root.val < p.val ? lowestCommonAncestorBST(root.right, p, q) : lowestCommonAncestorBST(root.left, p, q)); 
    }
