Unique Binary Search Trees II

    /**
     * LeetCode 95
     * Dynamic Programming.
     * @param n - integer
     * @return all structurally unique BST's (binary search trees) that store values 1...n.
     */
    public static List generateTrees(int n) {
        if (n < 1) {
            return new ArrayList();
        }
        return generateTreesForRange(1, n);
    }

    //Helper - construct sub trees for Range [low, high].
    private static List generateTreesForRange(int low, int high) {
        List rv = new ArrayList();
        
        if (low > high) {
            rv.add(null);
        }
        
        for (int i = low; i <= high; i++) {
            List leftTrees = generateTreesForRange(low, i - 1);
            List rightTrees = generateTreesForRange(i + 1, high);
            
            for (TreeNode leftTree : leftTrees) {
                for (TreeNode rightTree : rightTrees) {
                    TreeNode root = new TreeNode(i);
                    root.left = leftTree;
                    root.right = rightTree;
                    rv.add(root);
                }
            }
        }
        
        return rv;
    }
